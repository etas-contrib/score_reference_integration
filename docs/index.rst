--------------------

   import re

   all_release_notes = []

      if "wp__platform_sw_release_note" in need["realizes"]:
         all_release_notes.append(need)

   newest_release_note = max(all_release_notes, key=lambda s: int(re.search(r'v(\d+)', s["id"]).group(1)))
   results = [newest_release_note]
